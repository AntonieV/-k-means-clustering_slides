%Folie 4
\section{Ziel und Voraussetzungen}
\subsection{K-Means}
\begin{frame}
	\frametitle{K-Means}
	\begin{block}{Ziel}
		Für alle Cluster soll die Summe der quadratischen Abstände der Datenpunkte zu ihrem Clusterzentrum minimiert werden.
	\end{block}
	\vspace{0.7cm}		
	\begin{block}{Voraussetzungen} 
		\begin{itemize}
			\item Anzahl der Klassen muss im Voraus bekannt sein
			\item Daten stammen aus $ d $-dimensionalen Raum
			\item Datensatz besteht aus sphärischen, gut abgrenzbaren Datenpunktanhäufungen				
		\end{itemize}
	\end{block}
\end{frame}

%Folie 5
\section{Berechnung}
\subsection{K-Means}
\begin{frame}
		\frametitle{K-Means}			
		\begin{definition} [Summe der quadratischen Abstände \textbf{eines} Clusters]							
			\[ J(c_{j}) = \sum_{a_{i}\in c_{j}}\|a_{i} - c_{j}\|^2 = \sum_{a_{i}\in c_{j}}d^2(a_{i},c_{j}) \]
		\end{definition}
		\vspace{0.2cm}	
		\visible<2>{
		\begin{definition} [Summe der quadratischen Abstände \textbf{aller} Cluster]
			\[ J(C) = \sum_{j = 1}^k J(c_{j}) = \sum_{j = 1}^k\sum_{a_{i}\in c_{j}}d^2(a_{i},c_{j}) \]
		\end{definition}
	}
	\begin{columns} [T]
		{\footnotesize 		
			\begin{column}{0.1\textwidth}
			\end{column}			
			\begin{column}{0.25\textwidth}
				\vfill
				\visible<2>{
				$k$ \\ }
				\visible<1->{
				$n$ \\
				$C=\{c_{1},...,c_{k}\}$ \\				
				$A=\{a_{1},...a_{n}\}$ \\}				
			\end{column}
			\begin{column}{0.64\textwidth}
					\vfill
					\visible<2>{
					Clusteranzahl \\ }
					\visible<1->{
					Anzahl der Datenpunkte \\
					Menge der Vektoren der Clusterzentren \\					
					Menge der Datenpunkte \\}					
			\end{column}
		}				
		\end{columns}
\end{frame}

\section{Optimale Clusteranzahl}
\subsection{Basisalgorithmus nach Lloyd}
\begin{frame}
	\frametitle{Die optimale Clusteranzahl $k$ bestimmen}
	\vspace{-0.3cm}
	\begin{itemize}
		\begin{block}{Clusterradius}
			Maximale Distanz zwischen den Punkten eines Clusters und dem Clusterzentrum.
		\end{block}		
		\item[]<2->
		\onslide<2-3>
		\begin{block}{Clusterdurchmesser}
			Maximale Distanz zwischen zwei Punkten eines Clusters.
		\end{block}
		\vspace{-0.2cm}		
		\item[]<3> 
		\onslide<3>
		\begin{block}{}
			\begin{itemize}{\normalsize
				\item Solange $k$ der tatsächlichen Clusteranzahl entspricht, verändern sich Clusterradius bzw. -durchmesser nur geringfügig beim Hinzufügen von weiteren Punkten.
				\item Werden zwei Cluster zusammengefasst, die nicht zueinander passen, dann kommt es zu einem deutlichen Anstieg des Clusterradius bzw. -durchmessers.}
			\end{itemize}
		\end{block}
	\end{itemize}

\end{frame}

\section{Datenanalyse}
\subsection{BFR-Algorithmus}
\begin{frame}
	\frametitle{Datenanalyse}
	\begin{block}{}
		Für jedes DS und CS werden folgende Daten gespeichert:
		\begin{itemize}
			\item \textbf{\textit{N}}: Anzahl der Datenpunkte
			\visible<2->{
			\item \textbf{\textit{SUM}}: Vektor, dessen Elemente jeweils die Summe der Koordinaten einer Dimension von allen Punkten des Clusters repräsentieren. \\}
			\visible<3->{
			\item \textbf{\textit{SUMSQ}}: Vektor, dessen Elemente jeweils die Summe der Koordinatenquadrate einer Dimension von allen Punkten des Clusters repräsentieren. }
		\end{itemize}
		\vspace{0.5cm}
		\visible<4->{
		\textbf{Beispiel:} \\}
		\vspace{-1cm}
		\begin{equation*}
		\visible<4->{
		a=\left( \begin{array}{c}{a_{1}} \\{a_{2}}\end{array}\right), \,\, b=\left( \begin{array}{c}{b_{1}} \\ {b_{2}}\end{array}\right) \qquad
		\begin{aligned}}
		\visible<5->{
		&\Rightarrow  N=2 \\}
		\visible<6->{
		&\Rightarrow SUM=\left( \begin{array}{c}{a_{1}+b_{1}} \\  {a_{2}+b_{2}}\end{array}\right) \\}
		\visible<7>{
		&\Rightarrow SUMSQ=\left( \begin{array}{c}{a_{1}\,^2+b_{1}\,^2} \\  {a_{2}\,^2+b_{2}\,^2}\end{array}\right) \\}
		\end{aligned}
		\end{equation*}
	\end{block}
\end{frame}

%Folie 17
\section{Datenanalyse}
\subsection{BFR-Algorithmus}
\begin{frame}
	\frametitle{Datenanalyse}
	Aus $N$, $SUM$ und $SUMSQ$ können die Koordinaten des Clusterzentrums und die Standardabweichung berechnet werden:
	\visible<2->{
	\begin{block}{Clusterzentrum}
		Die $i$-te Koordinate des Clusterzentrums ergibt sich aus den Mittelwerten der Datenpunktkoordinaten:
		\[\frac{SUM_{i}}{N} \]
	\end{block}}
	\visible<3->{
	\begin{block}{Standardabweichung}
		Die Varianz $v$ in der $i$-ten Dimension kann aus den $SUM$- und $SUMSQ$-Vektoren ermittelt werden:
		\[ v_{i} = \frac{SUMSQ_{i}}{N}-\left(\frac{SUM_{i}}{N}\right)^2 \] 
		Daraus ergibt sich die Standardabweichung $s_{i} = \sqrt{v_{i}}$ 
	\end{block}	}
\end{frame}

%Folie 18
\section{Datenanalyse}
\subsection{BFR-Algorithmus}
\begin{frame}
	\frametitle{Datenanalyse}
	\visible<1-3>{
	Warum entsprechen die Koordinaten des Clusterzentrums den Mittelwerten der Datenpunktkoordinaten? \\ }
	\visible<2-3>{
	\vfill
	Seien die Koordinaten eines Clusterzentrums $ c $ also:
	\[c = \frac{1}{n}\sum_{i = 1}^n a_{i} = \bar a\]}
	\vfill
	\visible<3>{Dann gilt für die Summe der quadratischen Abstände aller  Punkte $a_{i}$ eines Clusters zu einem Punkt $x$: \\
	\[\sum_{i = 1}^n |a_{i} - x|^2\]}
	
	\vspace{-8cm}
	\begin{flalign*} %ermöglicht untereinanderstehende Struktur, z.B. der "="-Zeichen
	\visible<4->{
	\sum_{i = 1}^n |a_{i} - x|^2 }
	\visible<5->{
	& = \sum_{i = 1}^n |a_{i} - c + c - x|^2 \\ }
	\visible<6->{
	& = \sum_{i = 1}^n |a_{i} - c|^2 + 2(c - x)\cdot \sum_{i = 1}^n (a_{i} - c ) + \sum_{i = 1}^n |c - x|^2 \\ }
	\visible<7->{
	&= \sum_{i = 1}^n |a_{i} - c|^2 + 2(c - x)\cdot \sum_{i = 1}^n (a_{i} - \bar a ) + n|c - x|^2 \\ }
	\visible<8->{
	& = \sum_{i = 1}^n |a_{i} - c|^2 + 2(c - x)\cdot 0 + n|c - x|^2 \\ }
	\visible<9->{
	& = \sum_{i = 1}^n |a_{i} - c|^2 + n|c - x|^2 }
	\end{flalign*}
	\visible<10>{$\Rightarrow$ ist genau dann minimal, wenn $x=c$ und die Koordinaten von $c$ den \textbf{Mittelwerten} der Datenpunktkoordinaten entsprechen.}
\end{frame}

	\begin{itemize}
		\visible<2->{
			\item Erhöhen der Anzahl der Datenpunkte $\Rightarrow$ $N+1$}
		\visible<3->{
			\item Koordinaten des Punktes zum SUM-Vektor addieren}
		\visible<4->{
			\item Quadrate der Punktkoordinaten zum SUMSQ-Vektor addieren}
	\end{itemize}
	\vspace{0.8cm}
	\visible<5->{
		Hinzufügen eines neuen Datenpunktes zum \textbf{RS}:}
	\begin{itemize}
		\visible<6->{
			\item[] $\Rightarrow$ der Datenpunkt selbst wird gespeichert}
	\end{itemize}
	\vspace{0.8cm}
\end{block}

\end{frame}

				\visible<5->{
				\item[] $\quad\Rightarrow$ ggf. neues CS mit bereits vorhandenen CS zusammenfassen}
				\vspace{0.15cm}
				\visible<6->{
				\item Datenpunkt liegt isoliert:
				\item[] $\quad\Rightarrow$ Punkt im RS speichern}
					}
			\end{itemize}
		}
			\vspace{0.4cm} 
		\end{enumerate}
	\end{block}

\section{Distanzmaß}
\subsection{BFR-Algorithmus}
\begin{frame}[t]
\strut
	\frametitle{Distanzmaß}
		\vspace{-0.8cm}
	\begin{center}
		{\large \textbf{Wann liegt ein Punkt nahe genug an einem Clusterzentrum?}}%
	\end{center}

		\begin{enumerate}[\begingroup\color{black} 1\endgroup]
			\temporal<4>{
			\item Jedes DS und CS enthält:%
			\begin{itemize}
				{\normalsize 				
				\item Koordinaten des Clusterzentrums $c = (c_{1},...,c_{d})$
				\item Standardabweichungen für jede Dimension $s = (s_{1},...,s_{d})$}%
				\end{itemize}}
			{\item Metadaten liefern Clusterzentren und Standardabweichungen}%
			{\item Metadaten liefern Clusterzentren und Standardabweichungen}%	
		\vspace{0.5 cm}
		\temporal<2-3>{				
			\par			
			\textcolor{white}{Normalisierte (euklidische) Distanz $z$ für jede Dimension $i$:
			\[z_{i} = \frac{a_{i}-c_{i}}{s_{i}}   \textcolor{white}{=\frac{s_{i}}{s_{i}} = 1}\]}}%
		{
			\par
			\item Normalisierte (euklidische) Distanz $z$ für jede Dimension $i$:
			\[\;\;\;\;\;\;\;\;\;\;\;\;\;z_{i} = \frac{a_{i}-c_{i}}{s_{i}}   \textcolor{white}{=\frac{s_{i}}{s_{i}} = 1}\]}%
		{			
			\par
			\item $a_{i}$ liegt in jeder Dimension genau die \textcolor{blue}{\textbf{einfache Standardabweichung}} vom Clusterzentrum entfernt:
			\[z_{i} = \frac{a_{i}-c_{i}}{s_{i}}  \textcolor{blue}{=\frac{s_{i}}{s_{i}} = 1}\]}%
		\temporal<3-4>{
			\textcolor{white}{\textbf{Mahalanobis-Distanz}:
			\[MD = \sqrt{\sum_{i=1}^{d} z_{i}^2 } \textcolor{white}{=\sqrt{\sum_{i=1}^{d} 1^2 } = \sqrt{d}}\]}}%
			{\item \textbf{Mahalanobis-Distanz}:
			 
			\[\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;\;MD = \sqrt{\sum_{i=1}^{d} z_{i}^2 } \textcolor{white}{=\sqrt{\sum_{i=1}^{d} 1^2 } = \sqrt{d}}\]}%
			{\item \textbf{Mahalanobis-Distanz}:
				\[MD = \sqrt{\sum_{i=1}^{d} z_{i}^2 } \textcolor{blue}{=\sqrt{\sum_{i=1}^{d} 1^2 } = \sqrt{d}}\]}%
			\visible<6>{$\Rightarrow$ Grenzen für 2- bzw. 3-fache Standardabweichung: $2\sqrt{d}$ bzw. $3\sqrt{d}$}%		
	\end{enumerate}
\end{frame}

\section{Kritische Betrachtung}
\subsection{BFR-Algorithmus}
\begin{frame}
	\frametitle{Kritische Betrachtung des BFR-Algorithmus}
	\begin{columns}[T]
		\begin{column}{0.49\textwidth}
			\begin{center}
				\textbf{Vorteile}
			\end{center}
			\begin{itemize}
				\item für multivariate Daten geeignet
				\visible<2->{
				\item keine Iteration}
				\visible<3->{
				\item generieren multipler Lösungen}
				\visible<4->{
				\item durch Metadaten geringerer Speicherplatzbedarf}
			\end{itemize}
		\end{column}
		\begin{column}{0.49\textwidth}
			\begin{center}
				\textbf{Nachteile}
			\end{center}
			\begin{itemize}
				\visible<5->{
				\item einzelne Datenpunkte aus Metadaten nicht mehr extrahierbar}
				\visible<6->{
				\item für Hintergrundrauschen und Ausreißer anfällig}
				\visible<7->{
				\item strenges Kriterium der Normalverteilung in jeder Dimension}
			\end{itemize}
		\end{column}
	\end{columns}
\end{frame}
